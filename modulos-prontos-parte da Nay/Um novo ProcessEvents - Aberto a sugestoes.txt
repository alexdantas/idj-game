void GameManager::processEvents(){
    input = input->getInstance();

    if(input->isKeyHold(SDLK_LEFT)){
        cavaloObj->moveX(-10);
    }
    if(input->isKeyHold(SDLK_RIGHT)){
        cavaloObj->moveX(10);
    }
    if(input->isKeyHold(SDLK_UP)){
        cavaloObj->moveY(-10);
    }
    if(input->isKeyHold(SDLK_DOWN)){
        cavaloObj->moveY(10);
    }

    if(input->isKeyHold(SDLK_x)){
        addPlaneta();
    }

}
